SELECT W.Salary 
FROM Employee E
JOIN Work W ON E.Emp_name = W.Emp_name
WHERE E.Emp_name = 'Adam';

SELECT C.City FROM Work W JOIN Company C ON W.Cmp_name = C.Cmp_name WHERE W.Emp_name = 'Brooks';

SELECT S.*
FROM Sailors S JOIN Reserves R
ON S.Sid = R.Sid
WHERE R.Day = TO_DATE('09-MAY-1998', 'DD-MON-YYYY');

SELECT S.Sname, R.Day
FROM Sailors S JOIN Reserves R
ON S.Sid = R.Sid
WHERE R.Bid = 103;

SELECT S.Sname, S.Age
FROM Sailors S
JOIN Reserves R ON S.Sid = R.Sid
JOIN Boats B ON R.Bid = B.Bid
WHERE B.Color = 'Red' AND B.Bid =101;


SELECT S.* 
FROM Sailors S LEFT JOIN Reserves R ON S.Sid = R.Sid
WHERE R.Sid IS NULL;


SELECT DISTINCT  S.Sname
FROM Sailors S
JOIN Reserves R ON S.Sid = R.Sid
JOIN Boats B ON R.Bid = B.Bid
WHERE B.color IN ('Red', 'Green');

SELECT S.*, B.*
FROM Sailors S
JOIN Reserves R ON S.Sid = R.Sid
JOIN Boats B ON R.Bid = B.Bid
WHERE B.Bname = 'Interlake';

SELECT S.Rating, B.*
FROM Sailors S
JOIN Reserves R ON S.Sid = R.Sid
JOIN Boats B ON B.Bid = R.Bid
WHERE R.Day = TO_DATE('10-OCT-1998', 'DD-MON-YY');

FROM Sailors S

JOIN Boats B ON B.Bid = R.Bid
