SELECT E.Emp_name, C.City
FROM Employee E 
JOIN Work W ON E.Emp_name = W.Emp_name
JOIN Company C ON W.Cmp_name = C.Cmp_name;

FROM Employee E
JOIN Work W ON E.Emp_name = W.Emp_name
WHERE W.Salary > 15000;

SELECT P.Dscr, SUM(SD.Qty_order) AS Total_Qty
FROM Product P 
JOIN Salesorder_Details SD ON P.Pr_no = SD.Pr_no
GROUP BY P.Dscr;

FROM Product P 
JOIN Salesorder_Details SD ON P.Pr_no = SD.Pr_no
JOIN Salesorder S ON S.Od_no = SD.Od_no
JOIN Client C ON C.Cl_no = S.Cl_no
WHERE C.Name = 'Ivan Bayross';
