Q3

FROM Borrow B
JOIN Branch Br ON B.Bname = Br.Bname
JOIN Customers C ON B.Cname = C.Cname

Q13
SELECT SUM(S.Qty_order * S.Rate) AS Total_Amount
FROM Salesorder O
JOIN Salesorder_Details S ON O.Od_no = S.Od_no
WHERE TO_CHAR(O.O_dat,'MON') = 'JUN'
AND O.Bill = 'Y';

Q14
SELECT S.Od_no, P.Dscr, S.Qty_order
FROM Product P
JOIN Salesorder_Details S ON P.Pr_no = S.Pr_no
WHERE P.Dscr = 'Pull Overs'
AND S.Qty_order < 5;

Q15
SELECT P.Dscr, O.Od_no
FROM Client C
JOIN Salesorder O ON C.Cl_no = O.Cl_no
JOIN Salesorder_Details S ON O.Od_no = S.Od_no
JOIN Product P ON S.Pr_no = P.Pr_no
WHERE C.Name IN ('Ivan Bayross','Mamta Muzumdar');
